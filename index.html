Backend (Node.js - chuẩn crawl full chương) ### package.json ```json { "name": "wattpad-epub", "version": "1.0.0", "main": "server.js", "scripts": { "start": "node server.js" }, "dependencies": { "axios": "^1.6.0", "cheerio": "^1.0.0", "cors": "^2.8.5", "epub-gen": "^0.1.0", "express": "^4.18.2" } } ``` ### server.js ```js const express = require("express"); const axios = require("axios"); const cheerio = require("cheerio"); const EPUB = require("epub-gen"); const cors = require("cors"); const app = express(); app.use(cors()); app.use(express.json()); async function getChapterLinks(url) { const { data } = await axios.get(url, { headers: { "User-Agent": "Mozilla/5.0" } }); const $ = cheerio.load(data); let links = []; $("a[href*='/story/'][href*='/chapter/']").each((i, el) => { const link = "https://www.wattpad.com" + $(el).attr("href"); if (!links.includes(link)) links.push(link); }); return links.slice(0, 50); // limit tránh quá tải } async function getChapterContent(url) { const { data } = await axios.get(url, { headers: { "User-Agent": "Mozilla/5.0" } }); const $ = cheerio.load(data); const title = $("h1").text() || "Chapter"; let content = ""; $("p").each((i, el) => { content += `

${$(el).text()}

`; }); return { title, data: content }; } app.post("/download", async (req, res) => { try { const { url } = req.body; const { data } = await axios.get(url); const $ = cheerio.load(data); const bookTitle = $("h1").first().text() || "Wattpad Book"; const chapterLinks = await getChapterLinks(url); let chapters = []; for (let link of chapterLinks) { try { const chap = await getChapterContent(link); chapters.push(chap); console.log("✔", chap.title); } catch (e) { console.log("❌ lỗi chương", link); } } const option = { title: bookTitle, author: "Wattpad", content: chapters }; const filePath = `./book.epub`; await new EPUB(option, filePath).promise; res.download(filePath); } catch (err) { console.error(err); res.status(500).send("Lỗi server"); } }); app.listen(3000, () => console.log("http://localhost:3000")); ``` --- ## 🌐 2.
